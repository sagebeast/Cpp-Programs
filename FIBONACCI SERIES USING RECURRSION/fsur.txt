
//9. fibonacci series using recurssion;

#include<iostream>
using namespace std;
int fib(int n){
	if(n==1||n==2){
		return 1;
	}else{
		return fib(n-1)+fib(n-2);
	}
	
	
}

int main(){
	int n ;
	cout<<"Enter terms of fibonacci series"<<endl;
	cin>>n;
	for(int i=1;i<=n;i++){
		cout<<fib(i)<<" ";
	}
	
	
return 0;
}
